9. Add TapInteraction for cluster circles:
   ```swift
   TapInteraction(.layer("crime-cluster-circles")) { feature, context in
       if let clusterFeature = feature.feature {
           proxy.map?.getGeoJsonClusterLeaves(
               forSourceId: "crime-clusters",
               feature: clusterFeature,
               limit: 100,
               offset: 0
           ) { result in
               switch result {
               case .success(let leaves):
                   let crimes = leaves.features.compactMap { f -> CrimeDetail? in
                       guard let p = f.properties,
                             let cat = p["category"]?.string,
                             let desc = p["description"]?.string,
                             let date = p["date"]?.string,
                             let sev = p["severity"]?.string else { return nil }
                       return CrimeDetail(
                           category: cat, description: desc, date: date,
                           severity: CrimeSeverity(rawValue: sev) ?? .other
                       )
                   }
                   DispatchQueue.main.async { onClusterTap(crimes) }
               case .failure(let error):
                   AppLogger.map.error("Failed to get cluster leaves: \(error)")
               }
           }
       }
       return true
   }
   ```
